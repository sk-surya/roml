      - Change::VariableAdded { var, bounds, var_type } -> ModelOp::AddVariable { var, bounds, var_type }

      - Change::ConstraintActivityChanged { con, active } -> ModelOp::SetConstraintActive { con, active }

      - Change::ObjectiveSenseChanged { obj, old:_, new:_ } -> (skipped if not tracked by ModelOp — ModelOp doesn't have SetObjectiveSense)
      - Change::ParameterValueChanged { param, old:_, new } -> ModelOp::SetParameter { param, value: new }
